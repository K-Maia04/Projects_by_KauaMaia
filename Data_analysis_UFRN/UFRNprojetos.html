def tabela_frequencia(variable):
    table = pd.DataFrame(total_loan[variable].value_counts())
    table.columns = ["Quantidade"]
    table['percentual'] = round((table['Quantidade'] / table['Quantidade'].sum())*100,2)
    return table

## Funo para Criao de Boxplots, para facilitar a criao dos grficos.
def create_boxplot(dataset,x,y,title,subtitle):
    plt.figure(figsize=(14,8))
    sns.set_theme(style='darkgrid')
    ax = sns.boxplot(x = x , y = y , data= dataset)
    ax.set_xlabel(str(x),fontsize = 14)
    ax.set_ylabel(str(y),fontsize = 14)
    ax.yaxis.set_major_formatter(ticker.FuncFormatter(lambda x, p: format(int(x), ',').replace(',','.')))
    ax.set_title(title+"\n",size=18,loc='left',weight='bold')
    ax.text(s=subtitle,x=-0.5,y=max(dataset[y])*1.11,fontsize=16, ha='left',color='gray')
    plt.ylim(0,max(dataset[y])*1.1)

def loan_years(year1, year2):
    loan_y1 = loan_all[loan_all['Ano'] == year1].sort_values(by='Curso').reset_index(drop=True)
    loan_y2 = loan_all[loan_all['Ano'] == year2].sort_values(by='Curso').reset_index(drop=True)
    course_table = (((loan_y2['Quantidade'] - loan_y1['Quantidade']) / loan_y1['Quantidade']) * 100).round(2)
    course_table = pd.DataFrame({'Curso': loan_y1['Curso'], str(year2): course_table})
    course_table[str(year2)] = course_table[str(year2)].astype(str) + '%'
    return course_table
